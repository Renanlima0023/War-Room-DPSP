-- ============================================================ -- SCRIPT PARA CRIAR AS TABELAS LOJAS E RESPONSAVEIS -- ============================================================ -- ============================================================ -- 1. Tabela RESPONSAVEIS -- ============================================================ CREATE TABLE IF NOT EXISTS responsaveis ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), nome VARCHAR(100) NOT NULL, created_at TIMESTAMP DEFAULT NOW() ); -- ============================================================ -- 2. Tabela LOJAS -- ============================================================ CREATE TABLE IF NOT EXISTS lojas ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), nome VARCHAR(100) NOT NULL, codigo VARCHAR(20), responsavel_id UUID REFERENCES responsaveis(id) ON DELETE SET NULL, status VARCHAR(20) DEFAULT 'agendada', regional VARCHAR(50), cidade VARCHAR(100), endereco TEXT, pdvs INTEGER DEFAULT 0, comentarios INTEGER DEFAULT 0, data_agendada DATE, prioridade VARCHAR(20), subtipo VARCHAR(100), descricao TEXT, created_at TIMESTAMP DEFAULT NOW(), updated_at TIMESTAMP DEFAULT NOW() ); -- ============================================================ -- 3. Índices -- ============================================================ CREATE INDEX IF NOT EXISTS idx_lojas_responsavel ON lojas(responsavel_id); CREATE INDEX IF NOT EXISTS idx_lojas_status ON lojas(status); -- ============================================================ -- 4. Políticas RLS -- ============================================================ ALTER TABLE responsaveis ENABLE ROW LEVEL SECURITY; ALTER TABLE lojas ENABLE ROW LEVEL SECURITY; -- Remover políticas existentes DROP POLICY IF EXISTS "responsaveis_leitura" ON responsaveis; DROP POLICY IF EXISTS "lojas_leitura" ON lojas; DROP POLICY IF EXISTS "responsaveis_escrita" ON responsaveis; DROP POLICY IF EXISTS "lojas_escrita" ON lojas; -- Leitura (todos podem ler) CREATE POLICY "responsaveis_leitura" ON responsaveis FOR SELECT USING (true); CREATE POLICY "lojas_leitura" ON lojas FOR SELECT USING (true); -- Escrita (apenas autenticados) CREATE POLICY "responsaveis_escrita" ON responsaveis FOR ALL USING (auth.role() = 'authenticated'); CREATE POLICY "lojas_escrita" ON lojas FOR ALL USING (auth.role() = 'authenticated'); -- ============================================================ -- 6.
